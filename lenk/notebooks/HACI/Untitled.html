def local_threshold(T, Tmax=273.15, thresh_min=240, Tmin=220, spread=5, factor = 10):
    T_thresh = np.max((T + spread, T + spread + factor*(T - Tmax)/ (Tmin - Tmax),thresh_min))
    return T_thresh
